def delete(posicion)

  if posicion == 0 
  	pop_head
  elsif posicion == (@sz - 1)
      pop_tail
  elsif posicion < (@sz - 1) && posicion > 0
      iterator = @head
      posicion.times do |i|
      iterator = iterator.next
  end   

  iterator.next.prev = iterator.prev
  iterator.prev.next = iterator.next
  @sz = @sz - 1

  end
 end

def push_head(value)

  if value.class == Nodo
    added_node = value
  else
    added_node = Nodo.new(value)
  end

  added_node.prev = nil
  added_node.next = @head

  @head.prev = added_node unless @head.nil?
  @head = added_node

  @tail = added_node if @tail.nil? 
  @sz = @sz + 1
  return nil
  
end
